#include <stdio.h>
#include <string.h>

int main() {
	string digits[10][5] = {
		{"111","101","101","101","111"}, //0
		{"100","100","100","100","100"}, //1
		{"111","001","111","100","111"}, //2
		{"111","100","111","100","111"}, //3
		{"100","100","111","101","101"}, //4
		{"111","100","111","001","111"}, //5
		{"111","101","111","001","111"}, //6
		{"100","100","100","100","111"}, //7
		{"111","101","111","101","111"}, //8
		{"111","100","111","101","111"}  //9
	};

	FILE *fp = fopen("out.txt", "w");
	for (int k = 0; k < 10; k++) { //10^1 digit
		for (int l = 0; l < 10; l++) { //10^0 digit
			fprintf(fp, "{B11111110, B00000000, B%s0%s0, B%s0%s0, B%s0%s0, B%s0%s0, B%s0%s0, B00000000},\n", digits[l][0].c_str(), digits[k][0].c_str(), digits[l][1].c_str(), digits[k][1].c_str(), digits[l][2].c_str(), digits[k][2].c_str(), digits[l][3].c_str(), digits[k][3].c_str(), digits[l][4].c_str(), digits[k][4].c_str());
		}
	}
	fclose(fp);
	getchar();
	return 0;
}
